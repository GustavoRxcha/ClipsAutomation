# Dependências do ClipsAutomation
# Instale com: pip install -r requirements.txt

yt-dlp             # Download de vídeos do YouTube (robusto contra bot detection)
faster-whisper     # Transcrição de áudio com Whisper (versão otimizada em CTranslate2)
ffmpeg-python      # Wrapper Python para o FFmpeg
python-dotenv      # Leitura de variáveis de ambiente do arquivo .env
google-api-python-client  # Cliente da API do Google (YouTube Data API v3)
google-auth-oauthlib      # Fluxo OAuth2 para autenticação do Google
tiktok-uploader           # Upload de vídeos para o TikTok via cookies (opcional)
